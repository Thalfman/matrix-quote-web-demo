# Plan D — Client-Ready PDF Export

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add server-side PDF generation via WeasyPrint for a single saved or ad-hoc quote. Output is a 2–3 page, Matrix-branded document (cover, breakdown, assumptions) that sales can email to a client.

**Architecture:** New `backend/app/pdf.py` renders a Jinja2 HTML template to PDF with WeasyPrint. Two new endpoints live on the existing `quotes` router: `GET /api/quotes/{id}/pdf` (saved scenario) and `POST /api/quote/pdf` (ad-hoc from cockpit before save). No frontend dependency additions. Dockerfile picks up WeasyPrint's native prerequisites.

**Tech Stack:** WeasyPrint (BSD-3), Jinja2 (already a FastAPI transitive dep), FastAPI `StreamingResponse`, React fetch-blob download flow.

**Spec reference:** `docs/superpowers/specs/2026-04-17-estimator-cockpit-redesign-design.md` — Section 8 (PDF export).

**Prerequisites:**
- Plan A merged (prediction shape finalized).
- Plan C merged (saved-quotes endpoints and frontend list).

```python
# backend/app/pdf.py
"""Render a saved-quote Jinja template to a PDF bytes payload."""

from __future__ import annotations

from datetime import datetime
from pathlib import Path
from typing import Any

from jinja2 import Environment, FileSystemLoader, select_autoescape
from weasyprint import HTML

from .schemas_api import SavedQuote


TEMPLATES_DIR = Path(__file__).resolve().parent / "templates"


BUCKET_LABELS = {
    "mechanical":    "Mechanical",
    "electrical":    "Electrical",
    "controls":      "Controls",
    "robotics":      "Robotics",
    "assembly":      "Assembly",
    "shipping":      "Shipping",
    "install":       "Install",
    "startup":       "Startup",
    "engineering":   "Engineering",
    "project_mgmt":  "Project management",
    "documentation": "Documentation",
    "misc":          "Misc",
}

INPUT_LABELS = {
    "industry_segment":       "Industry",
    "system_category":        "System category",
    "automation_level":       "Automation level",
    "plc_family":             "PLC family",
    "hmi_family":             "HMI family",
    "vision_type":            "Vision type",
    "stations_count":         "Stations",
    "robot_count":            "Robots",
    "conveyor_length_ft":     "Conveyor length (ft)",
    "fixture_sets":           "Fixture sets",
    "part_types":             "Part types",
    "servo_axes":             "Servo axes",
    "complexity_score_1_5":   "Complexity",
}


def _format_hours(n: float) -> str:
    return f"{round(n):,}"


def _humanize_bucket(key: str) -> str:
    return BUCKET_LABELS.get(key, key.replace("_", " ").title())


def _render_env() -> Environment:
    env = Environment(
        loader=FileSystemLoader(str(TEMPLATES_DIR)),
        autoescape=select_autoescape(["html", "xml"]),
    )
    env.globals["format_hours"] = _format_hours
    env.globals["humanize_bucket"] = _humanize_bucket
    return env


def _input_rows(inputs: Any) -> list[tuple[str, str]]:
    data = inputs.model_dump() if hasattr(inputs, "model_dump") else dict(inputs)
    rows: list[tuple[str, str]] = []
    for key, label in INPUT_LABELS.items():
        if key in data and data[key] not in (None, ""):
            rows.append((label, str(data[key])))
    return rows


def render_quote_pdf(quote: SavedQuote, *, quote_number: str) -> bytes:
    env = _render_env()
    template = env.get_template("quote_pdf.html")
    html = template.render(
        quote=quote,
        quote_number=quote_number,
        prepared_on=datetime.utcnow().strftime("%b %d, %Y"),
        input_rows=_input_rows(quote.inputs),
    )
    pdf_bytes = HTML(
        string=html,
        base_url=str(TEMPLATES_DIR),
    ).write_pdf()
    return pdf_bytes
```

```python
from datetime import datetime

from ..pdf import render_quote_pdf


def _quote_number(created_at: datetime) -> str:
    return f"{created_at:%Y%m%d}-{created_at:%H%M}"


@router.get("/{quote_id}/pdf")
def get_quote_pdf(quote_id: str) -> Response:
    q = quotes_storage.get(quote_id)
    if q is None:
        raise HTTPException(status_code=404, detail="Quote not found")
    pdf_bytes = render_quote_pdf(q, quote_number=_quote_number(q.created_at))
    fname = f"Matrix-Quote-{q.project_name.replace(' ', '-')}-{q.created_at:%Y%m%d}.pdf"

        media_type="application/pdf",
        headers={"Content-Disposition": f'attachment; filename="{fname}"'},
    )
```

```ts
async function _streamDownload(resp: { data: Blob }, fallbackName: string) {
  const url = URL.createObjectURL(resp.data);
  const a = document.createElement("a");
  a.href = url;
  a.download = fallbackName;
  document.body.appendChild(a);
  a.click();
  a.remove();
  URL.revokeObjectURL(url);
}

export async function downloadScenarioPdf(id: string): Promise<void> {
  const resp = await api.get(`/quotes/${id}/pdf`, { responseType: "blob" });
  _streamDownload(resp as { data: Blob }, `Matrix-Quote-${id}.pdf`);
}

export async function downloadAdHocPdf(body: {
  name: string;
  project_name: string;
  client_name?: string | null;
  created_by: string;
  inputs: QuoteInput;
  prediction: QuotePrediction;
}): Promise<void> {
  const resp = await api.post("/quote/pdf", body, { responseType: "blob" });
  _streamDownload(resp as { data: Blob }, `Matrix-Quote-${body.project_name.replace(/\s+/g, "-")}.pdf`);
}
```
