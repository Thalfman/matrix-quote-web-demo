## Why this plan

This plan delivers UX-02 (the headline ask: hover a Complexity vs Hours bar → see which projects make up that bucket) plus the chart side of UX-03 (definitions on chart category-name affordances). The chart side of UX-03 is intentionally lightweight: each chart card already has an "eyebrow" header text and (for ComplexityVsHours) a Recharts `XAxis` `label`. This satisfies acceptance #3 with minimal surface area and matches CONTEXT note "wrapper if the wrapped element is not natively focusable".

## Risks / pitfalls

- **R1 — Recharts axis-label focus.** Recharts renders the `XAxis` `label` as plain SVG `<text>`, which is NOT focusable in any browser. Do NOT try to make it focusable by passing `tabIndex` — Recharts strips unknown props on label objects. Instead, add a `<GlossaryHelpIcon>` next to the existing card "eyebrow" header (the `<div className="eyebrow text-xs text-muted">...</div>` element), which IS focusable because we're rendering a real `<button>`. This is the pattern CONTEXT calls out: "introduce a focusable proxy".
- **R2 — Recharts `<CustomTooltip>` payload typing.** The component receives `payload[0].payload` which is the chart row — in our case `LevelRow`. After adding `topProjects` and `overflow` to `LevelRow`, the type needs no further changes since `TooltipPayloadEntry` is `{ payload?: LevelRow }`.
- **R3 — Tooltip footprint.** Adding 5 project rows + an overflow row to the existing "Projects / Avg hours / Range" tooltip risks vertical overflow for tall buckets. Constrain the project list to one line per project (truncate name with `text-overflow: ellipsis` and `max-width: 200px`), and use `tnum` for hours. The existing `TOOLTIP_STYLE` already constrains the outer card visually.
- **R4 — Sort + slice in `bucketByComplexity`.** Sort the per-bucket project list DESC by hours BEFORE slicing top 5. The remaining `topProjects.length - 5` becomes `overflow`. Mirror Phase 1's `BucketRow.projectCount` pattern: enrich the row with the data the consumer needs, don't recompute in `<CustomTooltip>`.
- **R5 — Empty buckets.** For complexity levels with zero projects, `topProjects` should be `[]` and `overflow` should be `0`. The chart already short-circuits the "no data" UI when `hasAny` is false; do not change that branch.
- **R6 — Existing test for ComplexityVsHours.** I checked: there is currently no `ComplexityVsHours.test.tsx`. T2 creates it new. There IS a `portfolioStats.test.ts` (modified in Phase 1); T1 extends that file's existing structure rather than creating a new test.
- **R7 — `ScatterPoint` consumers.** Search for `ScatterPoint` usages with `grep -n "ScatterPoint" frontend/src` before T1; they live in `portfolioStats.ts` (the type) and `ComplexityVsHours.tsx` (consumer). Adding an optional `projectId?: string` field is non-breaking; making it required is breaking. We make it REQUIRED for clarity. The single consumer (ComplexityVsHours) will be updated in T2 in the same wave.
- **R8 — `<TooltipProvider>` mounting.** Each card now uses one `<Tooltip>`. Wrap a single `<TooltipProvider>` AT THE TOP of `BusinessInsightsView.tsx`? NO — that's plan 02-04's territory if it touches BIV; for chart cards, mount a `<TooltipProvider>` inside each chart component (around the eyebrow row) so this plan stays self-contained. Multiple providers don't conflict — Radix supports nesting.

// frontend/src/pages/demo/business/ComplexityVsHours.tsx
import {
  Bar, BarChart, CartesianGrid, LabelList, ResponsiveContainer, Tooltip as RechartsTooltip, XAxis, YAxis,
} from "recharts";

import { Tooltip, TooltipProvider, GlossaryHelpIcon } from "@/components/Tooltip";
import {
  AXIS_LINE, AXIS_TICK, CHART_COLORS, DATA_LABEL, GRID_STYLE, TOOLTIP_CURSOR, TOOLTIP_STYLE,
} from "@/pages/insights/chartTheme";
import { ScatterPoint } from "./portfolioStats";

const fmtHours = new Intl.NumberFormat("en-US", { maximumFractionDigits: 0 });

// UX-02 — drill-down list cap. Top 5 projects per bucket; rest collapsed into "+N more".
const TOP_N = 5;

export type LevelRow = {
  level: number;
  count: number;
  avg: number;
  min: number;
  max: number;
  /** Top N projects in this bucket, sorted desc by hours. UX-02. */
  topProjects: { projectId: string; projectName: string; hours: number }[];
  /** Number of projects beyond the top N (for "+N more" row). 0 if bucket has ≤ TOP_N. */
  overflow: number;
};

export function bucketByComplexity(points: ScatterPoint[]): LevelRow[] {
  // Per-level accumulators retain full project triples so the tooltip can
  // render the drill-down list without recomputing from the raw points.
  const by: Record<number, { hours: number; projectId: string; projectName: string }[]> = {
    1: [], 2: [], 3: [], 4: [], 5: [],
  };
  for (const p of points) {
    const level = Math.round(p.complexity);
    if (level >= 1 && level <= 5) {
      by[level].push({ hours: p.hours, projectId: p.projectId, projectName: p.name });
    }
  }
  return [1, 2, 3, 4, 5].map((level) => {
    const entries = by[level];
    if (entries.length === 0) {
      return { level, count: 0, avg: 0, min: 0, max: 0, topProjects: [], overflow: 0 };
    }
    // Sort desc by hours so slice(0, TOP_N) is "the top".
    entries.sort((a, b) => b.hours - a.hours);
    const xs = entries.map((e) => e.hours);
    const sum = xs.reduce((a, b) => a + b, 0);
    const top = entries.slice(0, TOP_N).map((e) => ({
      projectId: e.projectId,
      projectName: e.projectName,
      hours: e.hours,
    }));
    const overflow = Math.max(0, entries.length - TOP_N);
    return {
      level,
      count: entries.length,
      avg: sum / entries.length,
      min: Math.min(...xs),
      max: Math.max(...xs),
      topProjects: top,
      overflow,
    };
  });
}

type TooltipPayloadEntry = { payload?: LevelRow };

function CustomTooltip({
  active,
  payload,
}: {
  active?: boolean;
  payload?: TooltipPayloadEntry[];
}) {
  if (!active || !payload?.length) return null;
  const row = payload[0]?.payload;
  if (!row) return null;
  const rowStyle = { display: "flex", justifyContent: "space-between", gap: 16 };
  const labelStyle = { color: CHART_COLORS.muted };
  return (
    <div style={{ ...TOOLTIP_STYLE, maxWidth: 280 }}>
      <div style={{ fontWeight: 600, marginBottom: 4 }}>Complexity {row.level}</div>
      <div style={rowStyle}>
        <span style={labelStyle}>Projects</span>
        <span className="tnum">{row.count}</span>
      </div>
      <div style={rowStyle}>
        <span style={labelStyle}>Avg hours</span>
        <span className="tnum">{fmtHours.format(row.avg)}</span>
      </div>
      {row.count > 0 && (
        <div style={rowStyle}>
          <span style={labelStyle}>Range</span>
          <span className="tnum">
            {fmtHours.format(row.min)}–{fmtHours.format(row.max)}
          </span>
        </div>
      )}
      {row.topProjects.length > 0 && (
        <div style={{ marginTop: 6, paddingTop: 6, borderTop: `1px solid ${CHART_COLORS.line}` }}>
          <div style={{ ...labelStyle, marginBottom: 2, fontSize: 11 }}>Projects in this bucket</div>
          <ul style={{ margin: 0, padding: 0, listStyle: "none" }}>
            {row.topProjects.map((p) => (
              <li
                key={p.projectId || p.projectName}
                style={{ display: "flex", justifyContent: "space-between", gap: 12, fontSize: 12 }}
              >
                <span
                  style={{
                    overflow: "hidden",
                    textOverflow: "ellipsis",
                    whiteSpace: "nowrap",
                    maxWidth: 180,
                  }}
                >
                  {p.projectName}
                </span>
                <span className="tnum" style={{ color: CHART_COLORS.muted }}>
                  {fmtHours.format(p.hours)}
                </span>
              </li>
            ))}
            {row.overflow > 0 && (
              <li style={{ fontSize: 11, color: CHART_COLORS.muted, marginTop: 2 }}>
                +{row.overflow} more
              </li>
            )}
          </ul>
        </div>
      )}
    </div>
  );
}

type Props = {
  data: ScatterPoint[];
};

export function ComplexityVsHours({ data }: Props) {
  const rows = bucketByComplexity(data);
  const hasAny = rows.some((r) => r.count > 0);

  return (
    <TooltipProvider delayDuration={200}>

            </Tooltip>
          </div>
          <div className="text-xs text-muted mono tnum">{data.length} projects</div>

        {!hasAny ? (
          <div className="text-sm text-muted">No data available.</div>
        ) : (
          <div className="flex-1 min-h-0">
            <ResponsiveContainer width="100%" height="100%">

                <CartesianGrid {...GRID_STYLE} vertical={false} />

                  dataKey="level"
                  tick={AXIS_TICK}
                  axisLine={AXIS_LINE}
                  tickLine={false}
                  label={{
                    value: "Complexity (1–5)",
                    position: "insideBottom",

                    style: {
                      fontSize: 11,
                      fill: CHART_COLORS.muted,
                      fontFamily: AXIS_TICK.fontFamily,
                    },
                  }}
                />
                <YAxis
                  tick={AXIS_TICK}
                  axisLine={AXIS_LINE}
                  tickLine={false}
                  tickFormatter={(v: number) => fmtHours.format(v)}

                  <LabelList

      pt({ projectId: "a", name: "Alpha", complexity: 3, hours: 100 }),
      pt({ projectId: "b", name: "Bravo", complexity: 3, hours: 300 }),
      pt({ projectId: "c", name: "Charlie", complexity: 3, hours: 200 }),
    ];
    const rows = bucketByComplexity(points);
    const lvl3 = rows.find((r) => r.level === 3);
    expect(lvl3).toBeDefined();
    expect(lvl3!.count).toBe(3);
    expect(lvl3!.topProjects.map((p) => p.projectId)).toEqual(["b", "c", "a"]);
    expect(lvl3!.topProjects.map((p) => p.hours)).toEqual([300, 200, 100]);
    expect(lvl3!.overflow).toBe(0);
  });

  it("caps topProjects at 5 and reports overflow", () => {
    const points: ScatterPoint[] = [];
    for (let i = 0; i < 12; i++) {
      points.push(pt({ projectId: `id-${i}`, name: `Proj ${i}`, complexity: 4, hours: 100 + i }));
    }
    const rows = bucketByComplexity(points);
    const lvl4 = rows.find((r) => r.level === 4)!;
    expect(lvl4.count).toBe(12);
    expect(lvl4.topProjects.length).toBe(5);
    expect(lvl4.overflow).toBe(7);
    // Top 5 are the 5 highest-hours items (id-11 .. id-7 in this seed).
    expect(lvl4.topProjects[0].projectId).toBe("id-11");
    expect(lvl4.topProjects[4].projectId).toBe("id-7");
  });

  it("ignores points with complexity outside 1..5", () => {
    const points = [
      pt({ complexity: 0, hours: 999 }),
      pt({ complexity: 6, hours: 999 }),
      pt({ projectId: "x", complexity: 2, hours: 50 }),
    ];
    const rows = bucketByComplexity(points);
    expect(rows.find((r) => r.level === 2)!.count).toBe(1);
    expect(rows.find((r) => r.level === 1)!.count).toBe(0);
    expect(rows.every((r) => r.level >= 1 && r.level <= 5)).toBe(true);
  });

  it("rounds fractional complexity to nearest integer level", () => {
    const points = [
      pt({ complexity: 2.4, hours: 10 }),
      pt({ complexity: 2.5, hours: 20 }),
    ];
    const rows = bucketByComplexity(points);
    // 2.4 → 2; 2.5 → 3 (Math.round half-to-even on .5 may go either way in
    // some engines, but JS Math.round always rounds half to +Inf → 3).
    expect(rows.find((r) => r.level === 2)!.count).toBe(1);
    expect(rows.find((r) => r.level === 3)!.count).toBe(1);
  });
});
