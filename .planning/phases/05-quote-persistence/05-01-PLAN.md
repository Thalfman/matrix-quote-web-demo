```typescript
import type { QuoteInput } from "@/api/types";

/** First 2-3 chars of discipline. Falls back to "Quote". */
export function deriveSalesBucket(values: QuoteFormValues): string {
  // D-PATTERNS: there is no `sales_bucket` field on QuoteFormValues today —
  // the closest proxy is the SYSTEM_CATEGORY mapping. For Phase 5 we use a
  // simple heuristic: ME if has_controls && !has_robotics, EE if has_controls
  // && has_robotics is irrelevant here, ME+EE if both. The exact letter is
  // claude-discretion (CONTEXT D-XX) so long as it's stable.
  const hasME = values.stations_count > 0 || values.has_controls;
  const hasEE = values.has_robotics || values.servo_axes > 0;
  if (hasME && hasEE) return "ME+EE";
  if (hasEE) return "EE";
  if (hasME) return "ME";
  return "Quote";
}

/** Format: "{salesBucket} {hourLabel} · {visionLabel} · {ISODate}". Capped at 80 chars. */
export function buildAutoSuggestedName(values: QuoteFormValues, estimatedHours: number): string {
  const bucket = deriveSalesBucket(values);
  const hours = `${Math.round(estimatedHours).toLocaleString()}h`;
  const visionLabel =
    !values.vision_type || values.vision_type === "None" ? "No vision" : values.vision_type;
  const date = new Date().toISOString().slice(0, 10);
  const candidate = `${bucket} ${hours} · ${visionLabel} · ${date}`;
  if (candidate.length <= 80) return candidate;
  // Truncate the visionLabel segment first, leaving date intact.
  const overrun = candidate.length - 80;
  const truncatedVision = visionLabel.slice(0, Math.max(0, visionLabel.length - overrun - 1)) + "…";
  return `${bucket} ${hours} · ${truncatedVision} · ${date}`;
}
```

```typescript
/**
 * Owns the IndexedDB connection for "matrix-quotes" — one DB per tab.
 * Exposes save/list/get/delete/getVersions/restoreVersion plus a
 * BroadcastChannel for cross-tab sync. Pattern-mirrors
 * frontend/src/demo/pyodideClient.ts (module-singleton + lazy bootstrap +
 * subscribe). Phase 5 D-01..D-18.
 */
import { openDB, type IDBPDatabase } from "idb";

import {
  savedQuoteSchema,
  type SavedQuote,
  type WorkflowStatus,
  type Workspace,
  type QuoteVersion,
} from "./savedQuoteSchema";
import type { QuoteFormValues } from "@/pages/single-quote/schema";
import type { UnifiedQuoteResult } from "@/demo/quoteResult";

// ---------------------------------------------------------------------------
// Constants
// ---------------------------------------------------------------------------

export const QUOTE_DB_NAME = "matrix-quotes";
export const QUOTE_STORE_NAME = "quotes";
export const QUOTE_DB_VERSION = 1;
export const BROADCAST_CHANNEL_NAME = "matrix-quotes";

// ---------------------------------------------------------------------------
// Types
// ---------------------------------------------------------------------------

export type StorageEvent =
  | { type: "save";    id: string; updatedAt: string }
  | { type: "delete";  id: string }
  | { type: "restore"; id: string; updatedAt: string };

export type StorageListener = (event: StorageEvent) => void;

export interface SaveSavedQuoteArgs {
  id?: string;
  name: string;
  workspace: Workspace;
  status?: WorkflowStatus;     // defaults to "draft" for brand-new
  formValues: QuoteFormValues;
  unifiedResult: UnifiedQuoteResult;
  compareInputs?: { humanQuotedByBucket: Record<string, number> };
  restoredFromVersion?: number;
}

// ---------------------------------------------------------------------------
// Module state (singleton — mirrors pyodideClient.ts:78-86)
// ---------------------------------------------------------------------------

let dbPromise: Promise<IDBPDatabase> | null = null;
let channel: BroadcastChannel | null = null;
const listeners = new Set<StorageListener>();

// ---------------------------------------------------------------------------
// Channel bootstrap (lazy — created on first use)
// ---------------------------------------------------------------------------

function getChannel(): BroadcastChannel {
  if (!channel) {
    channel = new BroadcastChannel(BROADCAST_CHANNEL_NAME);
    channel.onmessage = (e: MessageEvent<StorageEvent>) => {
      // T-05-04: consumers must treat broadcast as cache-invalidate signal,
      // never as state. We re-broadcast to local listeners as-is; the
      // subscriber (Plan 04 hook) re-reads from IDB.
      for (const fn of listeners) fn(e.data);
    };
  }
  return channel;
}

function broadcast(evt: StorageEvent): void {
  // BroadcastChannel does NOT fire same-tab listeners (D-PATTERNS gotcha #7).
  // Notify local listeners first, then post to other tabs.
  for (const fn of listeners) fn(evt);
  getChannel().postMessage(evt);
}

// ---------------------------------------------------------------------------
// Lazy DB bootstrap (mirrors ensurePyodideReady pattern)
// ---------------------------------------------------------------------------

export function ensureDbReady(): Promise<void> {
  if (!dbPromise) {
    dbPromise = openDB(QUOTE_DB_NAME, QUOTE_DB_VERSION, {
      upgrade(db, oldVersion) {
        // schemaVersion bump: see CONTEXT.md D-18 — Phase 6/7 forward compat.
        if (oldVersion < 1) {
          const store = db.createObjectStore(QUOTE_STORE_NAME, { keyPath: "id" });
          store.createIndex("updatedAt", "updatedAt");
          store.createIndex("status", "status");
          store.createIndex("workspace", "workspace");
        }
      },
    }).catch((err: Error) => {
      dbPromise = null;  // Allow retry on next call.
      throw err;
    });
  }
  return dbPromise.then(() => undefined);
}

async function db(): Promise<IDBPDatabase> {
  if (!dbPromise) await ensureDbReady();
  return dbPromise!;
}

// ---------------------------------------------------------------------------
// Public API
// ---------------------------------------------------------------------------

export async function listSavedQuotes(): Promise<SavedQuote[]> {
  const handle = await db();
  // Read via updatedAt index, reverse for newest-first.
  const tx = handle.transaction(QUOTE_STORE_NAME, "readonly");
  const all = await tx.store.index("updatedAt").getAll();
  await tx.done;
  return all.sort((a, b) => (a.updatedAt < b.updatedAt ? 1 : -1));
}

export async function getSavedQuote(id: string): Promise<SavedQuote | null> {
  const handle = await db();
  const rec = await handle.get(QUOTE_STORE_NAME, id);
  if (!rec) return null;
  // T-05-05: validate on read. If schemaVersion is in the future, throw.
  return savedQuoteSchema.parse(rec);
}

export async function saveSavedQuote(args: SaveSavedQuoteArgs): Promise<SavedQuote> {
  const handle = await db();
  const now = new Date().toISOString();

  // T-05-01: revalidate name + payload at the boundary.
  let record: SavedQuote;
  if (args.id) {
    const existing = await getSavedQuote(args.id);
    if (!existing) {
      throw new Error(`saveSavedQuote: no quote found with id ${args.id}`);
    }
    const lastVersion = existing.versions[existing.versions.length - 1];
    const inputsChanged =
      JSON.stringify(lastVersion.formValues) !== JSON.stringify(args.formValues) ||
      JSON.stringify(lastVersion.unifiedResult) !== JSON.stringify(args.unifiedResult);

    const versions: QuoteVersion[] = inputsChanged
      ? [
          ...existing.versions,
          {
            version: lastVersion.version + 1,
            savedAt: now,
            statusAtTime: args.status ?? existing.status,
            formValues: args.formValues,
            unifiedResult: args.unifiedResult,
            ...(args.restoredFromVersion !== undefined && { restoredFromVersion: args.restoredFromVersion }),
            ...(args.compareInputs && { compareInputs: args.compareInputs }),
          },
        ]
      : existing.versions;

    record = {
      ...existing,
      name: args.name.trim(),
      status: args.status ?? existing.status,
      updatedAt: now,
      versions,
      salesBucket: deriveSalesBucketFromValues(args.formValues),
      visionLabel: args.formValues.vision_type || "No vision",
      materialsCost: args.formValues.estimated_materials_cost ?? 0,
    };
  } else {
    // Brand-new quote — create v1.
    record = {
      id: crypto.randomUUID(),
      schemaVersion: 1,
      name: args.name.trim(),
      workspace: args.workspace,
      status: args.status ?? "draft",
      createdAt: now,
      updatedAt: now,
      versions: [
        {
          version: 1,
          savedAt: now,
          statusAtTime: args.status ?? "draft",
          formValues: args.formValues,
          unifiedResult: args.unifiedResult,
          ...(args.compareInputs && { compareInputs: args.compareInputs }),
        },
      ],
      salesBucket: deriveSalesBucketFromValues(args.formValues),
      visionLabel: args.formValues.vision_type || "No vision",
      materialsCost: args.formValues.estimated_materials_cost ?? 0,
    };
  }

  // Defense-in-depth re-validation (T-05-01).
  const validated = savedQuoteSchema.parse(record);

  await handle.put(QUOTE_STORE_NAME, validated);
  broadcast({ type: "save", id: validated.id, updatedAt: validated.updatedAt });
  return validated;
}

export async function deleteSavedQuote(id: string): Promise<void> {
  const handle = await db();
  await handle.delete(QUOTE_STORE_NAME, id);
  broadcast({ type: "delete", id });
}

export async function getVersions(id: string): Promise<QuoteVersion[]> {
  const rec = await getSavedQuote(id);
  return rec ? rec.versions : [];
}

export async function setStatus(id: string, status: WorkflowStatus): Promise<SavedQuote> {
  const handle = await db();
  const existing = await getSavedQuote(id);
  if (!existing) throw new Error(`setStatus: no quote with id ${id}`);
  const updated: SavedQuote = {
    ...existing,
    status,
    updatedAt: new Date().toISOString(),
  };
  const validated = savedQuoteSchema.parse(updated);
  await handle.put(QUOTE_STORE_NAME, validated);
  broadcast({ type: "save", id: validated.id, updatedAt: validated.updatedAt });
  return validated;
}

export async function restoreVersion(
  id: string,
  version: number,
): Promise<{ formValues: QuoteFormValues }> {
  const rec = await getSavedQuote(id);
  if (!rec) throw new Error(`restoreVersion: no quote with id ${id}`);
  const target = rec.versions.find((v) => v.version === version);
  if (!target) throw new Error(`restoreVersion: no version ${version} in quote ${id}`);
  // D-06: NON-DESTRUCTIVE. Caller will re-save with restoredFromVersion to
  // commit the new fork.
  return { formValues: target.formValues };
}

export function subscribe(fn: StorageListener): () => void {
  listeners.add(fn);
  // Lazy-init the channel on first subscribe.
  getChannel();
  return () => {
    listeners.delete(fn);
  };
}

// ---------------------------------------------------------------------------
// Helpers
// ---------------------------------------------------------------------------

function deriveSalesBucketFromValues(values: QuoteFormValues): string {
  // Mirrors deriveSalesBucket in savedQuoteSchema.ts but kept private here
  // to avoid a circular import at the type-only boundary.
  const hasME = values.stations_count > 0 || values.has_controls;
  const hasEE = values.has_robotics || values.servo_axes > 0;
  if (hasME && hasEE) return "ME+EE";
  if (hasEE) return "EE";
  if (hasME) return "ME";
  return "Quote";
}
```
